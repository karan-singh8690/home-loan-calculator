   531→/** Build the initial CalcState from defaults overlaid with URL params. */
   532→function buildInitialState(url: ReturnType<typeof parseUrlState>): CalcState {
   533→  const base: CalcState = {
   534→    ...DEFAULT_STATE,
   535→    startDate: url.start ? fromMonthInputValue(url.start) : new Date(),
   536→  };
   537→  if (url.loan !== undefined) base.loanAmount = url.loan;
   538→  if (url.rate !== undefined) base.annualRate = url.rate;
   539→  if (url.years !== undefined) base.termYears = url.years;
   540→  if (url.months !== undefined) base.termExtraMonths = url.months;
   541→  // Phase 4: emi=0 (or blank) means "auto-calculate". Only treat a positive
   542→  // value as a manual override.
   543→  if (url.emi !== undefined && url.emi > 0) {
   544→    base.monthlyPayment = url.emi;
   545→    base.paymentIsManual = true;
   546→  }
   547→  if (url.opMonthly !== undefined) base.overpaymentMonthly = url.opMonthly;
   548→  if (url.opLump !== undefined) base.overpaymentLumpSum = url.opLump;
   549→  if (url.opAnnual !== undefined) base.overpaymentAnnual = url.opAnnual;
   550→  if (url.opStart !== undefined) base.overpaymentStartMonth = url.opStart;
   551→  if (url.timing) base.overpaymentTiming = url.timing;
   552→  // Phase 4: `mode` is the prepayment type (monthly/lump/both).
   553→  if (url.mode) base.overpaymentType = url.mode;
   554→  // `engineMode` is the tenure/emi strategy (V1 always tenure).
   555→  if (url.engineMode) base.prepaymentMode = url.engineMode;
   556→  if (url.inputMode) base.inputMode = url.inputMode;
   557→
   558→  // Pre-fill bank default rate if the tool is a bank view.
   559→  if (url.tool && isViewId(url.tool)) {
   560→    const bank = getBank(url.tool);
   561→    if (bank && url.rate === undefined) base.annualRate = bank.defaultRate;
   562→  }
   563→
   564→  // Compute an initial EMI if not manually set.
   565→  if (!base.paymentIsManual) {
   566→    const t = base.termYears * 12 + base.termExtraMonths;
   567→    const payment = calcMonthlyPayment(
   568→      base.loanAmount,
   569→      base.annualRate / 100 / 12,
   570→      t
   571→    );
   572→    base.monthlyPayment = Math.round(payment * 100) / 100;
   573→  }
   574→  base.lumpSums = base.lumpSums ?? [];
   575→  return base;
   576→}

   638→/** Build a plain-text summary of the results for "Copy results". */
   639→function buildCopyText(
   640→  r: ReturnType<typeof calculateMortgage>,
   641→  state: CalcState,
   642→  view: ViewMeta
   643→): string {
   644→  const lines: string[] = [];
   645→  lines.push(`${view.title.toUpperCase()}`);
   646→  lines.push("=".repeat(view.title.length));
   647→  lines.push("");
   648→  lines.push("YOUR HOME LOAN");
   649→  lines.push(`  Loan amount:        ${formatCurrency(state.loanAmount)}`);
   650→  lines.push(`  Interest rate:      ${state.annualRate}% p.a.`);
   651→  lines.push(`  Tenure:             ${formatDuration(totalTermMonths(state))}`);
   652→  lines.push(`  Monthly EMI:        ${formatCurrency(state.monthlyPayment)}`);
   653→  lines.push("");
   654→  if (view.showPrepayment) {
   655→    lines.push("PREPAYMENT STRATEGY");
   656→    lines.push(`  Mode:               ${state.prepaymentMode === "emi" ? "Reduce EMI" : "Reduce tenure"}`);
   657→    const parts: string[] = [];
   658→    if (state.overpaymentMonthly > 0)
   659→      parts.push(`${formatCurrency(state.overpaymentMonthly)}/month`);
   660→    if (state.overpaymentLumpSum > 0)
   661→      parts.push(`${formatCurrency(state.overpaymentLumpSum)} lump sum`);
   662→    if (state.overpaymentAnnual > 0)
   663→      parts.push(`${formatCurrency(state.overpaymentAnnual)}/year`);
   664→    if (state.lumpSums.length > 0) {
   665→      for (const ls of state.lumpSums as LumpSumPrepayment[]) {
   666→        parts.push(`${formatCurrency(ls.amount)} in month ${ls.month}`);
   667→      }
   668→    }
   669→    lines.push(
   670→      `  Strategy:           ${parts.join(" + ") || "none"} (from month ${state.overpaymentStartMonth}, applied at ${state.overpaymentTiming} of month)`
   671→    );
   672→    lines.push("");
   673→  }
   674→  lines.push("RESULTS");
   675→  lines.push(`  Original payoff:    ${formatDateLong(r.originalPayoffDate)}`);
   676→  lines.push(`  New payoff:         ${formatDateLong(r.newPayoffDate)}`);
   677→  if (r.prepaymentMode === "emi") {
   678→    lines.push(`  EMI reduced by:     ${formatCurrency(r.emiReduction)}/month`);
   679→    lines.push(`  Final EMI:          ${formatCurrency(r.finalEMI)}`);
   680→  } else {
   681→    lines.push(
   682→      `  Time saved:         ${r.monthsSaved > 0 ? formatDuration(r.monthsSaved) : "—"}`
   683→    );
   684→  }
   685→  lines.push(`  Interest saved:     ${formatCurrency(r.totalInterestSaved)}`);
   686→  if (r.prepaymentMode === "tenure") {
   687→    lines.push(
   688→      `  EMIs saved:         ${formatCurrency(r.totalPaymentsSaved)} (${r.monthsSaved} EMIs)`
   689→    );
   690→  }
   691→  lines.push("");
   692→  lines.push(`Calculated with HomeLoan Calc`);
   693→  return lines.join("\n");
   694→}

   696→function toMonthInputValue(d: Date): string {
   697→  const m = String(d.getMonth() + 1).padStart(2, "0");
   698→  const y = d.getFullYear();
   699→  return `${y}-${m}`;
   700→}
   701→
   702→function fromMonthInputValue(v: string): Date {
   703→  const [y, m] = v.split("-").map(Number);
   704→  return new Date(y, (m || 1) - 1, 1);
   705→}
