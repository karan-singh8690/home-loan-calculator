   635→/**
   636→ * Look up a single translated string. Falls back to the English value
   637→ * if the key is missing for the requested locale (defensive — should
   638→ * not normally happen because both locales are exhaustive).
   639→ */
   640→export function getTranslation(
   641→  locale: Locale,
   642→  key: keyof TranslationKey,
   643→): string {
   644→  const dict = translations[locale] ?? translations.en;
   645→  const value = dict[key];
   646→  if (value === undefined || value === null) {
   647→    return translations.en[key] ?? String(key);
   648→  }
   649→  return value;
   650→}
