  1824→/**
  1825→ * Returns the FAQs registered against a logical content type (e.g. "prepayment",
  1826→ * "emi", "sbi", "hdfc"). Falls back to the general prepayment FAQ set when the
  1827→ * requested type is not recognised so callers always get something to render.
  1828→ */
  1829→export function getHindiFaqsForType(type: string): HindiFaq[] {
  1830→  const t = type.toLowerCase();
  1831→
  1832→  const filterBy = (predicate: (q: string) => boolean): HindiFaq[] =>
  1833→    HINDI_FAQ_LIBRARY.filter((f) => predicate(f.question.toLowerCase()));
  1834→
  1835→  if (t === "prepayment" || t === "general" || t === "") {
  1836→    return filterBy(
  1837→      (q) =>
  1838→        q.includes("प्रीपेमेंट") ||
  1839→        q.includes("पार्ट पेमेंट") ||
  1840→        q.includes("जल्दी") ||
  1841→        q.includes("फोरक्लोज़र"),
  1842→    );
  1843→  }
  1844→
  1845→  if (t === "emi" || t === "interest-saving" || t === "interest") {
  1846→    return filterBy(
  1847→      (q) =>
  1848→        q.includes("emi") ||
  1849→        q.includes("ब्याज") ||
  1850→        q.includes("रिटर्न") ||
  1851→        q.includes("अवधि"),
  1852→    );
  1853→  }
  1854→
  1855→  if (t === "reduce-emi-vs-tenure" || t === "emi-vs-tenure") {
  1856→    return filterBy(
  1857→      (q) =>
  1858→        q.includes("emi") ||
  1859→        q.includes("अवधि") ||
  1860→        q.includes("रिटर्न"),
  1861→    );
  1862→  }
  1863→
  1864→  if (t === "sbi") {
  1865→    return filterBy((q) => q.includes("sbi"));
  1866→  }
  1867→  if (t === "hdfc") {
  1868→    return filterBy((q) => q.includes("hdfc"));
  1869→  }
  1870→  if (t === "icici") {
  1871→    return filterBy((q) => q.includes("icici"));
  1872→  }
  1873→  if (t === "axis") {
  1874→    return filterBy((q) => q.includes("axis"));
  1875→  }
  1876→  if (t === "bank") {
  1877→    return filterBy((q) =>
  1878→      ["sbi", "hdfc", "icici", "axis"].some((b) => q.includes(b)),
  1879→    );
  1880→  }
  1881→
  1882→  if (t === "balance-transfer" || t === "refinance") {
  1883→    return filterBy(
  1884→      (q) =>
  1885→        q.includes("बैलेंस ट्रांसफर") ||
  1886→        q.includes("रिफाइनेंस") ||
  1887→        q.includes("फ्लोटिंग") ||
  1888→        q.includes("फिक्स्ड"),
  1889→    );
  1890→  }
  1891→
  1892→  if (t === "cibil") {
  1893→    return filterBy((q) => q.includes("cibil"));
  1894→  }
  1895→
  1896→  if (t === "tax") {
  1897→    return filterBy((q) => q.includes("टैक्स") || q.includes("कर"));
  1898→  }
  1899→
  1900→  if (t === "foreclosure") {
  1901→    return filterBy((q) => q.includes("फोरक्लोज़र") || q.includes("शुल्क"));
  1902→  }
  1903→
  1904→  // Default: full library so callers always get usable FAQ content.
  1905→  return HINDI_FAQ_LIBRARY;
  1906→}
